import json import re import sys import tkinter as tk from datetime import datetime from pathlib import Path from tkinter import Tk, simpledialog, messagebox import certifi import requests try: import truststore truststore.inject_into_ssl() except Exception: pass FIREBASE_DB_URL = "https://office-attendance-bfc67-default-rtdb.firebaseio.com" APP_DIR = Path(sys.executable).resolve().parent if getattr(sys, "frozen", False) else Path(__file__).resolve().parent CONFIG_FILE = APP_DIR / "config.json" def clean_key(text): return re.sub(r"[^a-zA-Z0-9_-]", "_", text.strip()) def today_text(): return datetime.now().strftime("%Y-%m-%d") def is_weekday(): return datetime.now().weekday() < 5 def load_name(root): if CONFIG_FILE.exists(): with open(CONFIG_FILE, "r", encoding="utf-8") as file: data = json.load(file) if data.get("name"): return data["name"] name = simpledialog.askstring("Office Attendance", "Enter your name:", parent=root) if not name: sys.exit() name = name.strip() with open(CONFIG_FILE, "w", encoding="utf-8") as file: json.dump({"name": name}, file, indent=2) return name def record_attendance(name, working_from): today = today_text() safe_name = clean_key(name) url = f"{FIREBASE_DB_URL}/attendance/{today}/{safe_name}.json" payload = { "name": name, "inOffice": working_from in ["GC", "BNE"], "workingFrom": working_from, "date": today, "recordedAt": datetime.now().isoformat(timespec="seconds") } response = requests.put(url, json=payload, timeout=10, verify=certifi.where()) response.raise_for_status() def get_monitor_bounds(root): try: import ctypes monitors = [] monitor_enum_proc = ctypes.WINFUNCTYPE( ctypes.c_int, ctypes.c_ulong, ctypes.c_ulong, ctypes.POINTER(ctypes.c_long * 4), ctypes.c_double ) def callback(hmonitor, hdc, rect, data): left, top, right, bottom = rect.contents monitors.append((left, top, right - left, bottom - top)) return 1 ctypes.windll.user32.EnumDisplayMonitors( 0, 0, monitor_enum_proc(callback), 0 ) if monitors: return monitors except Exception: pass return [(0, 0, root.winfo_screenwidth(), root.winfo_screenheight())] def show_fade_message(root, text): monitors = get_monitor_bounds(root) popups = [] for monitor_x, monitor_y, monitor_w, monitor_h in monitors: popup = tk.Toplevel(root) popup.overrideredirect(True) popup.attributes("-topmost", True) popup.attributes("-alpha", 0.9) width = 280 height = 72 x = monitor_x + monitor_w - width - 30 y = monitor_y + monitor_h - height - 80 popup.geometry(f"{width}x{height}+{x}+{y}") popup.configure(bg="#222222") label = tk.Label( popup, text=text, fg="white", bg="#222222", font=("Segoe UI", 13) ) label.pack(expand=True, fill="both") popups.append(popup) def fade(alpha=0.9): alpha -= 0.05 if alpha <= 0: for popup in popups: try: popup.destroy() except Exception: pass root.destroy() return for popup in popups: try: popup.attributes("-alpha", alpha) except Exception: pass root.after(80, fade, alpha) root.after(900, fade) def ask_working_from(root, name): selected = {"value": None} popup = tk.Toplevel(root) popup.title("Office Attendance") popup.resizable(False, False) popup.attributes("-topmost", True) width = 330 height = 180 x = int((popup.winfo_screenwidth() - width) / 2) y = int((popup.winfo_screenheight() - height) / 2) popup.geometry(f"{width}x{height}+{x}+{y}") tk.Label( popup, text=f"Hi {name}, working from?", font=("Segoe UI", 12) ).pack(pady=(24, 16)) button_frame = tk.Frame(popup) button_frame.pack(pady=8) def choose(value): selected["value"] = value popup.destroy() for option in ["Home", "GC", "BNE"]: tk.Button( button_frame, text=option, width=8, command=lambda value=option: choose(value) ).pack(side="left", padx=6) popup.protocol("WM_DELETE_WINDOW", popup.destroy) root.wait_window(popup) return selected["value"] def main(): root = Tk() root.withdraw() root.update_idletasks() if not is_weekday(): root.destroy() return name = load_name(root) working_from = ask_working_from(root, name) if not working_from: root.destroy() return try: record_attendance(name, working_from) show_fade_message(root, f"Attendance recorded: {working_from}") root.mainloop() except Exception as error: messagebox.showerror("Error", f"Could not record attendance:\n\n{error}", parent=root) root.destroy() if __name__ == "__main__": main()
